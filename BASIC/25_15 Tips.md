## 1. Optional Unwrapping

```swift
var optional1: Int?
var optional2: Int?

// 1- Ugly Code
func unwrap() {
  if let o1 = optional1 {
    if let o2 = optional2 {
      //reached
    }
  }
}

// 2- Clean Code
func unwrap() {
  guard let o1 = optional1, o2 = optional2 else { return }
  // reached 
}
```
두 개(o1, o2)가 모두 nil이 아닐 경우 //reached, 하나라도 nil일 경우 return문

## 2. 삼항연산자

```swift
// clean code
let myBool = true
let testBool = myBool ? false : true // false
let testInt = myBool ? 16 : 8 // 16

// dirty code 
if myBool {
  testInt = 16
}else {
  testInt = 8
}
```

## 3. Generic

```swift
// dirty code
var strings = ["A", "B", "C", "D", "E"]
var integers = [1, 2, 3, 4, 5]

func printStrings(_ array: [String]){
  for s in array {
    print(s)
  }
}

func printIntegers(_ array: [Int]){
  for i in array {
    print(s)
  }
}

printStrings(strings) 
printIntegers(integers)  

// clean code
func printArray<T>(_ array: [T]){
  for item in array {
    print(item)
  }
}
```
